\begin{forthel}
    [synonym point/-s]

    \begin{signature}
        A point is a notion.
    \end{signature}

    Let $x, x', x''$ stand for points.

    \begin{axiom}
        $x = x$.
    \end{axiom}

    \begin{axiom}
        $x'' = x''$.
    \end{axiom}

    \begin{definition}
        $x' ~  x''$ iff $x' = x''$.
    \end{definition}
\end{forthel}
