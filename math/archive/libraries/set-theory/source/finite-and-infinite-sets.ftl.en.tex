\begin{smodule}{finite-and-infinite-sets.ftl}
\importmodule[libraries/set-theory]{omega-is-cardinal.ftl}

\begin{definition}[forthel]
  Let $x$ be a set.
  $x$ is \emph{finite} iff $\SETcard{x} \ORDless \ORDomega$.
\end{definition}

\begin{definition}[forthel]
  Let $x$ be a set.
  $x$ is \emph{infinite} iff $x$ is not finite.
\end{definition}

\begin{proposition}[forthel]
  Let $x$ be a set.
  Then $x$ is finite iff $\SETcard{x} \Eq n$ for some $n \Elem \ORDomega$.
\end{proposition}

\begin{proposition}[forthel]
  Let $x$ be a set.
  Then $x$ is infinite iff $\SETcard{x} \ORDgeq \ORDomega$.
\end{proposition}
\begin{proof}[forthel]
  $\SETcard{x} \ORDgeq \ORDomega$ iff $\SETcard{x} \ORDnless \ORDomega$.
\end{proof}

\begin{proposition}[forthel]
  $\ORDomega$ is infinite.
\end{proposition}
