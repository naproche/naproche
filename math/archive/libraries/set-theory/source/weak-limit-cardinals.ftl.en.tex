\begin{definition}[forthel,id=SET_THEORY_06_8945210236547450]
  A \emph{weak limit cardinal} is a cardinal number $\lambda$ such that $\lambda$ is not a successor cardinal and $\lambda \neq \ORDzero$.
\end{definition}
