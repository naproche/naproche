\symdecl*{Powerset Axiom}

\begin{axiom}[forthel,title=Powerset,name=Powerset Axiom]
  Let $x$ be a set.
  Then $\SETpow(x)$ is a set.
  Let the powerset of $x$ stand for $\SETpow(x)$.
\end{axiom}
