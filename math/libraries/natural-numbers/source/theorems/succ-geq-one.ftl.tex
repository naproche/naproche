\begin{proposition}[forthel,id=SuccGeqOneProp]
  Let $n$ be a natural number.
  Then $n \plus \one \geq \one$.
\end{proposition}
\begin{proof}[forthel]
  \begin{case}{$n \eq \zero$.} \end{case}

  \begin{case}{$n \neq \zero$.}
    Then $n \gtr \zero$.
    Hence $n \plus \one \gtr \zero \plus \one \eq \one$.
  \end{case}
\end{proof}
