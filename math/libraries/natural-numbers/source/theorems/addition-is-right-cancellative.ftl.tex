\begin{smodule}{addition-is-right-cancellative.ftl}

\importmodule[libraries/natural-numbers]{definitions?addition.ftl}
\importmodule[libraries/natural-numbers]{axioms?succ-is-injective.ftl}

\begin{proposition}[forthel,id=AddIsRightCancellativeProp]
  Let $n, m$ be natural numbers.
  For all $k \in \Nat$ if $n \plus k \eq m \plus k$ then $n \eq m$.
\end{proposition}
\begin{proof}[forthel,method=induction on $k$]
  Let $k \in \Nat$.

  \begin{case}{$k \eq \zero$.} \end{case}

  \begin{case}{$k \neq \zero$.} \end{case}
\end{proof}
