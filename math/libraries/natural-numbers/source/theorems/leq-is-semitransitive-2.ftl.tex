\begin{proposition}[forthel,id=LeqIsSemitransitiveIIProp]
  Let $n, m, k$ be natural numbers.
  If $n \less m \leq k$ then $n \less k$.
\end{proposition}
\begin{proof}[forthel]
  Assume $n \less m \leq k$.
  If $m \eq k$ then $n \less k$.
  If $m \less k$ then $n \less k$.
\end{proof}
