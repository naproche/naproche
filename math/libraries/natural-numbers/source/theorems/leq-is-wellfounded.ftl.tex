\begin{proposition}[forthel,id=LeqIsWellfoundedProp]
  Let $A$ be a nonempty subclass of $\Nat$.
  Then there exists a $m \elem A$ such that $m \leq n$ for all $n \elem A$.
\end{proposition}
\begin{proof}[forthel]
  Assume the contrary.
  Then for each $n \elem A$ there exists a $m \elem A$ such that $m \less n$.
  We can show by induction on $n$ that every $n \elem \Nat$ is less than any element of $A$. Obvious.
\end{proof}
