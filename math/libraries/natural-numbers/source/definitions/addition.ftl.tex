\begin{smodule}[title=Addition]{addition.ftl}

\importmodule[libraries/natural-numbers]{signatures?natural-numbers.ftl}

\symdef{plus}{\mathbin{\comp+}}

\begin{signature}[forthel,id=PlusSig]
  Let $n, m$ be natural numbers.
  $\emph{n \plus m}$ is a natural number.

  Let the \emph{sum of $n$ and $m$} stand for $n \plus m$.
\end{signature}

\begin{axiom}[forthel,id=RightPlusZeroAx]
  Let $n$ be a natural number.
  Then $n \plus \zero \eq n$.
\end{axiom}

\begin{axiom}[forthel,id=RightPlusSuccAx]
  Let $n, m$ be natural numbers.
  Then $n \plus \succ(m) \eq \succ(n \plus m)$.
\end{axiom}
