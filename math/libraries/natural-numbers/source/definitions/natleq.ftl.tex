\importmodule[libraries/natural-numbers]{definitions?leq.ftl}

\symdef{Natleq}[args=1]{\comp{\mathbb{N}}_{\comp\natleq#1}}

\begin{definition}[forthel,id=NatleqDef,for=Natleq]
  Let $n$ be a natural number.
  $\emph{\Natleq{n}} = \{ k \in \Nat \mid k \leq n \}$.
\end{definition}
