\symdef{Natgeq}[args=1]{\comp{\mathbb{N}}_{\comp\natgeq#1}}

\begin{definition}[forthel,id=NatgeqDef,for=Natgeq]
  Let $n$ be a natural number.
  $\emph{\Natgeq{n}} = \{ k \in \Nat \mid k \geq n \}$.
\end{definition}
