\importmodule[libraries/maps]{definitions?im.ftl}
\importmodule[libraries/maps]{definitions?restr.ftl}

\begin{proposition}[forthel,id=RestOfInjectMapIsBijecToImProp]
  Let $A, B$ be classes and $f \frominto{A}{B}$ and $X \subclass A$.
  Then $f \rest X$ is a bijection between $X$ and $\im{f}{X}$.
\end{proposition}
