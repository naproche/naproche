\begin{smodule}[title=The Image Under Identity Maps]{im-under-id.ftl}

\importmodule[libraries/classes]{definitions?subclasses.ftl}
\importmodule[libraries/maps]{definitions?im.ftl}
\importmodule[libraries/maps]{definitions?id.ftl}

\begin{proposition}[forthel,id=ImOfSubclassUnderIdProp]
  Let $A$ be a class and $X \subclass A$.
  Then $\im{\id{A}}{X} \eq X$.
\end{proposition}
