\begin{proposition}[forthel,id=InvIsBijectProp]
  Let $A, B$ be classes and $f \fromonto{A}{B}$.
  Assume that $f$ is invertible.
  Then $\inv{f}$ is a bijection between $B$ and $A$.
\end{proposition}
\begin{proof}[forthel]
  $\inv{f}$ is a surjective map from $B$ onto $A$.
  $\inv{f}$ is invertible.
  Hence $\inv{f}$ is injective.
  Therefore $\inv{f}$ is a bijection between $B$ and $A$.
\end{proof}
