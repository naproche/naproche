\begin{proposition}[forthel,id=PreimOfCodomProp]
  Let $A, B$ be classes and $f \fromto{A}{B}$.
  Then $\preim{f}{B} \eq A$.
\end{proposition}
\begin{proof}[forthel]
  We have $\preim{f}{B} = \{ a \in A \mid f(a) \elem B \}$.
  $f(a) \elem B$ for all $a \elem A$.
  Hence the thesis.
\end{proof}
