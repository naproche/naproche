\begin{proposition}[forthel,id=PreimOfIntersectProp]
  Let $A, B$ be classes and $f \fromto{A}{B}$ and $Y, Y' \subclass B$.
  Then $\preim{f}{Y \intersect Y'} \eq \preim{f}{Y} \intersect \preim{f}{Y'}$.
\end{proposition}
\begin{proof}[forthel]
  Let us show that $\preim{f}{Y \intersect Y'} \subclass \preim{f}{Y} \intersect \preim{f}{Y'}$.
    Let $a \elem \preim{f}{Y \intersect Y'}$.
    Then $f(a) \elem Y \intersect Y'$.
    Hence $f(a) \elem Y$ and $f(a) \elem Y'$.
    Thus $a \elem \preim{f}{Y}$ and $a \elem \preim{f}{Y'}$.
    Therefore $a \elem \preim{f}{Y} \intersect \preim{f}{Y'}$.
  End.

  Let us show that $\preim{f}{Y} \intersect \preim{f}{Y'} \subclass \preim{f}{Y \intersect Y'}$.
    Let $a \elem \preim{f}{Y} \intersect \preim{f}{Y'}$.
    Then $a \elem \preim{f}{Y}$ and $a \elem \preim{f}{Y'}$.
    Hence $f(a) \elem Y$ and $f(a) \elem Y'$.
    Thus $f(a) \elem Y \intersect Y'$.
    Therefore $a \elem \preim{f}{Y \intersect Y'}$.
  End.
\end{proof}
