\begin{proposition}[forthel,id=CompIsAssociativeProp]
  Let $A, B, C, D$ be classes and $f \fromto{A}{B}$ and $g \fromto{B}{C}$ and $h \fromto{C}{D}$.
  Then $h \compo (g \compo f) \eq (h \compo g) \compo f$.
\end{proposition}
\begin{proof}[forthel]
  $h \compo (g \compo f)$ and $(h \compo g) \compo f$ are maps from $A$ to $D$.

  Let us show that $(h \compo (g \compo f))(a) \eq ((h \compo g) \compo f)(a)$ for all $a \elem A$.
    Let $a \elem A$.
    Then $(h \compo (g \compo f))(a)
      \eq h((g \compo f)(a))
      \eq h(g(f(a)))
      \eq (h \compo g)(f(a))
      \eq ((h \compo g) \compo f)(a)$.
  End.

  Hence $h \compo (g \compo f) \eq (h \compo g) \compo f$.
\end{proof}
