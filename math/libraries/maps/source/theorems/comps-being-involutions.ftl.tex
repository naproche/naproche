\begin{proposition}[forthel,id=CompInvolProp]
  Let $A$ be a class and $f, g$ be involutions on $A$.
  Then $g \compo f$ is an involution on $A$ iff $g \compo f \eq f \compo g$.
\end{proposition}
\begin{proof}[forthel]
  \begin{case}{$g \compo f$ is an involution on $A$.}
    Then $\inv{g \compo f}
      \eq \inv{f} \compo \inv{g}
      \eq f \compo g$.
    Indeed $f$ and $g$ are invertible and surjective onto $A$.
  \end{case}

  \begin{case}{$g \compo f \eq f \compo g$.}
    $f \compo f$, $f \compo g$ and $f \compo g$ are maps on $A$.
    Hence
    \[  (g \compo f) \compo (g \compo f)       \]
    \[    \eq (g \compo f) \compo (f \compo g)   \]
    \[    \eq ((g \compo f) \compo f) \compo g   \]
    \[    \eq (g \compo (f \compo f)) \compo g   \]
    \[    \eq (g \compo \id{A}) \compo g       \]
    \[    \eq g \compo g                       \]
    \[    \eq \id{A}.                        \]
    Thus $g \compo f$ is selfinverse.
  \end{case}
\end{proof}
