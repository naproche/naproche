\begin{proposition}[forthel,id=PreimOfImOfSubclassSuperclassProp]
  Let $A, B$ be classes and $f \fromto{A}{B}$ and $X \subclass A$.
  Then $\preim{f}{\im{f}{X}} \supclass X$.
\end{proposition}
\begin{proof}[forthel]
  Let $a \elem X$.
  Then $f(a) \elem \im{f}{X}$.
  Hence $a \elem \preim{f}{\im{f}{X}}$.
  Indeed $\im{f}{X} \subclass B$.
\end{proof}
