\importmodule[libraries/maps]{definitions?involutions.ftl}

\begin{proposition}[forthel,id=IdIsInvolProp]
  Let $A$ be a class.
  $\id{A}$ is an involution on $A$.
\end{proposition}
\begin{proof}[forthel]
  We have $\id{A} \comp \id{A} \eq \id{A}$.
  Hence $\id{A}$ is selfinverse.
\end{proof}
