\begin{proposition}[forthel,id=ValueOfInvProp]
  Let $A, B$ be classes and $f \fromonto{A}{B}$ and $b \elem B$.
  Assume that $f$ is invertible.
  Then $f(\inv{f}(b)) \eq b$.
\end{proposition}
\begin{proof}[forthel]
  We have
  $f(\inv{f}(b))
    \eq (f \compo \inv{f})(b)
    \eq \id{B}(b)
    \eq b$.
\end{proof}
