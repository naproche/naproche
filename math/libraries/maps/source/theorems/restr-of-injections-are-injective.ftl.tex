\begin{proposition}[forthel,id=RestrOfInjectMapIsInjectProp]
  Let $A, B$ be classes and $f \frominto{A}{B}$ and $X \subclass A$.
  Then $f \rest X$ is injective.
\end{proposition}
\begin{proof}[forthel]
  Let $a, a' \elem X$.
  Assume $(f \rest X)(a) \eq (f \rest X)(a')$.
  Then $f(a) \eq f(a')$.
  Hence $a \eq a'$.
\end{proof}
