\begin{proposition}[forthel,id=ImOfPreimOfSubclassSubclassProp]
  Let $A, B$ be classes and $f \fromto{A}{B}$ and $Y \subclass B$.
  Then $\im{f}{\preim{f}{Y}} \subclass Y$.
\end{proposition}
