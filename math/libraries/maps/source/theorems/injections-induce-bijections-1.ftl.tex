\begin{proposition}[forthel,id=InjectIsBijectToRangeProp]
  Let $A, B$ be classes and $f \frominto{A}{B}$.
  Then $f$ is a bijection between $A$ and $\range(f)$.
\end{proposition}
\begin{proof}[forthel]
  $f$ is injective and surjects onto $\range(f)$.
  Hence $f$ is a bijection between $A$ and $\range(f)$.
\end{proof}
