\importmodule[libraries/maps]{definitions?range.ftl}

\begin{proposition}[forthel,id=ImOfDomProp]
  Let $A, B$ be classes and $f \fromto{A}{B}$.
  Then $\im{f}{A} \eq \range(f)$.
\end{proposition}
