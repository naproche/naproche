\begin{smodule}{const-is-surjective.ftl}

\importmodule[libraries/classes]{definitions?empty-classes.ftl}
\importmodule[libraries/classes]{definitions?singleton-class.ftl}
\importmodule[libraries/maps]{definitions?const.ftl}
\importmodule[libraries/maps]{definitions?surjections-between-classes.ftl}

\begin{proposition}[forthel,id=ConstIsSurjectProp]
  Let $A$ be a nonempty class and $a$ be an object.
  $\const{A}{a}$ is a surjective map from $A$ onto $\singleton{a}$.
\end{proposition}
