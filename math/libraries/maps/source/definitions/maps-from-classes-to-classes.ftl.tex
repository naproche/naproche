\importmodule[libraries/preliminaries]{preliminaries.ftl}

\symdef{fromto}[args=2]{\mathbin{\comp:}#1\mathbin{\comp\rightarrow}#2}

\begin{definition}[forthel,id=MapFromToDef,for=fromto]
  Let $A, B$ be classes.
  A \emph{map from $A$ to $B$} is a map $f$ such that $\dom(f) \eq A$ and $f(a) \elem B$ for each $a \elem A$.

  Let $\emph{f \fromto{A}{B}}$ stand for $f$ is a map from $A$ to $B$.

  Let a \emph{function from $A$ to $B$} stand for a map from $A$ to $B$ that is a function.
\end{definition}
