\begin{smodule}[title=Restriction]{restr.ftl}

\importmodule[libraries/classes]{definitions?subclasses.ftl}

\symdef{rest}{\mathbin{\comp\restriction}}

\begin{definition}[forthel,id=RestrDef,for=rest]
  Let $f$ be a map and $X \subclass \dom(f)$.
  $\emph{f \rest X}$ is the map defined on $X$ such that $(f \rest X)(a) \eq f(a)$ for all $a \elem X$.

  Let the \emph{restriction of $f$ to $X$} stand for $f \rest X$.
\end{definition}
