\begin{definition}[forthel,id=MapOfDef]
  Let $A$ be a class.
  A \emph{map of $A$} is a map $f$ such that $\dom(f) \eq A$.

  Let a \emph{function of $A$} stand for a map of $A$ that is a function.
\end{definition}
