\begin{smodule}[title=Preimage]{preim.ftl}

\importmodule[libraries/preliminaries]{preliminaries.ftl}

\symdef{preim}[args=2]{#1^{\comp{-1}}\comp[#2\comp]}

\begin{definition}[forthel,id=PreimDef]
  Let $f$ be a map and $B$ be a class.
  $\emph{\preim{f}{B}} = \{ a \in \dom(f) \mid f(a) \elem B \}$.

  Let the \emph{preimage of $B$ under $f$} stand for $\preim{f}{B}$.
\end{definition}
