\importmodule[libraries/maps]{definitions?maps-from-classes-to-classes.ftl}

\symdef{funspace}[args=2]{\comp[#1\mathbin{\comp\rightarrow}#2\comp]}

\begin{definition}[forthel,id=FunspaceDef,for=funspace]
  Let $A, B$ be classes.
  $\emph{\funspace{A}{B}}$ is the class of all functions from $A$ to $B$.
\end{definition}
