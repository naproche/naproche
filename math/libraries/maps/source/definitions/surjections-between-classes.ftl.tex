\symdef{fromonto}[args=2]{\mathbin{\comp:}#1\mathbin{\comp\twoheadrightarrow}#2}

\begin{definition}[forthel,id=SurjectFromToDef,for=fromonto]
  Let $A, B$ be classes.
  A \emph{surjective map from $A$ to $B$} is a map of $A$ that is surjective onto $B$.

  Let a \emph{surjective map from $A$ onto $B$} stand for a surjective map from $A$ to $B$.
  Let $\emph{f \fromonto{A}{B}}$ stand for $f$ is a surjective map from $A$ onto $B$.
\end{definition}
