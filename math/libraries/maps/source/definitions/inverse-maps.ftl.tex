\begin{definition}[forthel,id=AnInvDef]
  Let $f$ be a map.
  An \emph{inverse of $f$} is a map $g$ from $\range(f)$ to $\dom(f)$ such that $f(a) \eq b \iff g(b) \eq a$ for all $a \elem \dom(f)$ and all $b \elem \dom(g)$.
\end{definition}
