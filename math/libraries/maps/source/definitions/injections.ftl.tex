\importmodule[libraries/maps]{definitions?maps-from-classes-to-classes.ftl}

\symdef{frominto}[args=2]{\mathbin{\comp:}#1\mathbin{\comp\hookrightarrow}#2}

\begin{definition}[forthel,id=InjectDef,for=frominto]
  Let $f$ be a map.
  $f$ is \emph{injective} iff for all $a, a' \elem \dom(f)$ if $f(a) \eq f(a')$ then $a \eq a'$.

  Let $\emph{f \frominto{A}{B}}$ stand for $f$ is an injective map from $A$ to $B$.
\end{definition}
