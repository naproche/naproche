\importmodule[libraries/preliminaries]{preliminaries.ftl}

\symdef{range}{\comp\textrm{range}}

\begin{definition}[forthel,id=RangeDef,for=range]
  Let $f$ be a map.
  $\emph{\range(f)} = \{ f(a) \mid a \elem \dom(f) \}$.

  Let the \emph{range of $f$} stand for $\range(f)$.
\end{definition}
