\importmodule[libraries/preliminaries]{preliminaries.ftl}

\symdef{intersect}{\mathbin{\comp\cap}}

\begin{definition}[forthel,id=IntersectDef,for=intersect]
  Let $A, B$ be classes.
  $\emph{A \intersect B} = \{ x \mid x \elem A\text{ and }x \elem B \}$.

  Let the \emph{intersection of $A$ and $B$} stand for $A \intersect B$.
\end{definition}
