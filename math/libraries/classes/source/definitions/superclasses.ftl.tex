\importmodule[libraries/preliminaries]{preliminaries.ftl}

\symdef{supclass}{\mathrel{\comp\supseteq}}

\begin{definition}[forthel,id=SuperclassDef,for=supclass]
  Let $A$ be a class.
  A \emph{superclass of $A$} is a class $B$ such that every element of $A$ is an
  element of $B$.

  Let $\emph{B \supclass A}$ stand for $B$ is a superclass of $A$.
\end{definition}
