\importmodule[libraries/classes]{definitions?subclasses.ftl}

\symdef{propsubclass}{\mathrel{\comp\subsetneq}}

\begin{definition}[forthel,id=ProperSubclassDef,for=propsubclass]
  Let $A$ be a class.
  A \emph{proper subclass of $A$} is a subclass $B$ of $A$ such that $B \neq A$.

  Let $\emph{B \propsubclass A}$ stand for $B$ is a proper subclass of $A$.
\end{definition}
