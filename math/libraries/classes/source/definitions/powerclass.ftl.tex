\begin{smodule}[title=Powerclass]{powerclass.ftl}

\importmodule[libraries/sets]{definitions?subsets.ftl}

\symdef{pow}{\comp{\mathcal{P}}}

\begin{definition}[forthel,id=PowerclassDef]
  Let $A$ be a class.
  $\emph{\pow(A)}$ is $\{ x \mid x\text{ is a subset of }A \}$.

  Let the \emph{powerclass of $A$} stand for $\pow(A)$.
\end{definition}
