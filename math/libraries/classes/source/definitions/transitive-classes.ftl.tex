\importmodule[libraries/sets]{definitions?subsets.ftl}

\begin{definition}[forthel,id=TransClassDef]
  Let $A$ be a class.
  $A$ is \emph{transitive} iff every element of $A$ is a subset of $A$.
\end{definition}
