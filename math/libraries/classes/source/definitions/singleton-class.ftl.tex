\symdef{singleton}[args=1]{\comp{\{}#1\comp{\}}}

\begin{definition}[forthel,id=SingletonClassDef,for=singleton]
  Let $a$ be an object.
  $\emph{\singleton{a}} = \{ x \mid x \eq a \}$.

  Let the \emph{singleton class of $a$} stand for $\singleton{a}$.
  Let a \emph{singleton class} stand for an entity $x$ such that $x \eq \singleton{a}$ for some object $a$.
\end{definition}
