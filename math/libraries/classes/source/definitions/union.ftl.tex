\symdef{union}{\mathbin{\comp\cup}}

\begin{definition}[forthel,id=UnionDef,for=union]
  Let $A, B$ be classes.
  $\emph{A \union B} = \{ x \mid x \elem A\text{ or }x \elem B \}$.

  Let the \emph{union of $A$ and $B$} stand for $A \union B$.
\end{definition}
