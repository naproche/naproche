\importmodule[libraries/classes]{definitions?empty-class.ftl}

\begin{proposition}[forthel,id=DifferenceOfEqualClassesProp]
  Let $A$ be a class.
  Then $A \diff A \eq \emptyclass$.
\end{proposition}
\begin{proof}[forthel]
  $A \diff A$ has no elements.
  Indeed $A \diff A = \{ x \mid x \elem A\text{ and }x \nelem A \}$.
  Hence the thesis.
\end{proof}
