\begin{proposition}[forthel,id=SubclassOfUnionProp]
  Let $A, B$ be classes.
  Then $A \subclass A \union B$.
\end{proposition}
\begin{proof}[forthel]
  Let $x \elem A$.
  Then $x \elem A$ or $x \elem B$.
  Hence $x \elem A \union B$.
\end{proof}
