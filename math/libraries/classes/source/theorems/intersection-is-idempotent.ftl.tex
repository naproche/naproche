\begin{proposition}[forthel,id=IntersectIsIdempotent]
  Let $A$ be a class.
  Then $A \intersect A \eq A$.
\end{proposition}
\begin{proof}[forthel]
  $A \intersect A = \{ x \mid x \elem A\text{ and }x \elem A \}$.
  Hence $A \intersect A = \{ x \mid x \elem A \}$.
  Thus $A \intersect A \eq A$.
\end{proof}
