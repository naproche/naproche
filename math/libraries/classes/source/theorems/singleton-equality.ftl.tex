\importmodule[libraries/classes]{definitions?singleton-class.ftl}

\begin{proposition}[forthel,id=SingletonEqualityProp]
  Let $a, a'$ be objects.
  If $\singleton{a} \eq \singleton{a'}$ then $a \eq a'$.
\end{proposition}
