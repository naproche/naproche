\begin{proposition}[forthel,id=UnionIsIdempotentProp]
  Let $A$ be a class.
  Then $A \union A \eq A$.
\end{proposition}
\begin{proof}[forthel]
  $A \union A = \{x \mid x \elem A\text{ or }x \elem A \}$.
  Hence $A \union A = \{ x \mid x \elem A \}$.
  Thus $A \union A \eq A$.
\end{proof}
