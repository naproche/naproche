\begin{proposition}[forthel,id=ProductWithIntersectOnRhsProp]
  Let $A, B, C$ be classes.
  Then $A \prod (B \intersect C) \eq (A \prod B) \intersect (A \prod C)$.
\end{proposition}
\begin{proof}[forthel]
  Let us show that $A \prod (B \intersect C) \subclass (A \prod B) \intersect (A \prod C)$.
    Let $x \elem A \prod (B \intersect C)$.
    Take $a \elem A$ and $b \elem B \intersect C$ such that $x \eq (a, b)$.
    Then $b \elem B$ and $b \elem C$.
    Hence $x \elem A \prod B$ and $x \elem A \prod C$.
    Thus $x \elem (A \prod B) \intersect (A \prod C)$.
  End.

  Let us show that $((A \prod B) \intersect (A \prod C)) \subclass A \prod (B \intersect C)$. %!
    Let $x \elem (A \prod B) \intersect (A \prod C)$.
    Then $x \elem A \prod B$ and $x \elem A \prod C$.
    Take objects $y, z$ such that $x \eq (y, z)$.
    Then $y \elem A$ and ($z \elem B$ and $z \elem C$).
    Hence $x \elem A \prod (B \intersect C)$.
  End.
\end{proof}
