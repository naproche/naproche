\begin{proposition}[forthel,id=SuperclassOfIntersectProp]
  Let $A, B$ be classes.
  Then $A \intersect B \subclass A$.
\end{proposition}
\begin{proof}[forthel]
  Let $x \elem A \intersect B$.
  Then $x \elem A$ and $x \elem B$.
  Hence $x \elem A$.
\end{proof}
