\begin{smodule}[title=Inclusion is Reflexive]{inclusion-is-reflexive.ftl}

\importmodule[libraries/classes]{definitions?subclasses.ftl}

\begin{proposition}[forthel,id=InclusionIsReflProp]
  Let $A$ be a class.
  Then $A \subclass A$.
\end{proposition}
\begin{proof}[forthel]
  Every element of $A$ is contained in $A$.
  Therefore $A \subclass A$.
\end{proof}
