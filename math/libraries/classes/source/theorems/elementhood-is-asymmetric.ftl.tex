\importmodule[libraries/sets]{axioms?set-induction.ftl}

\begin{proposition}[forthel,id=ElementhoodIsAsymmetricProp]
  There exist no classes $A, B$ such that $A \elem B$ and $B \elem A$.
\end{proposition}
\begin{proof}[forthel]
  We can show by induction on $X$ that for any class $X$ there exists no class $Y$ such that
  $X \elem Y$ and $Y \elem X$. Obvious.
\end{proof}
