\begin{proposition}[forthel,id=InclusionIsAntisymProp]
  Let $A, B$ be classes.
  If $A \subclass B$ and $B \subclass A$ then $A \eq B$.
\end{proposition}
\begin{proof}[forthel]
  Assume $A \subclass B$ and $B \subclass A$.
  Then every element of $A$ is contained in $B$ and every element of $B$ is contained in $A$.
  Hence $A \eq B$.
\end{proof}
