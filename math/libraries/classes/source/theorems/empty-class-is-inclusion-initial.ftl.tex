\begin{proposition}[forthel,id=EveryClassIncludesEmptyClassProp]
  Let $A$ be a class.
  Then $\emptyclass \subclass A$.
\end{proposition}
\begin{proof}[forthel]
  $\emptyclass$ has no elements.
  Hence every element of $\emptyclass$ is contained in $A$.
\end{proof}
