\importmodule[libraries/maps]{definitions?id.ftl}

\begin{proposition}[forthel,id=EquinumerosityIsReflexiveProp]
  Let $A$ be a class.
  Then $A$ is equinumerous to $A$.
\end{proposition}
\begin{proof}[forthel]
  $\id{A}$ is a bijection between $A$ and $A$.
\end{proof}
