\begin{smodule}[title=Symmetric Difference of Equal Classes]{symdiff-of-equal-classes.ftl}

\importmodule[libraries/classes]{definitions?symdiff.ftl}
\importmodule[libraries/classes]{definitions?empty-class.ftl}

\begin{proposition}[forthel,id=SymdiffOfEqualClassesProp]
  Let $A$ be a class.
  Then $A \symdiff A \eq \emptyclass$.
\end{proposition}
\begin{proof}[forthel]
  $A \symdiff A
    \eq (A \union A) \diff (A \intersect A)
    \eq A \diff A
    \eq \emptyclass$.
\end{proof}
