\symdef{Sym}[args=1]{\comp{\mathfrak{S}}_{#1}}

\begin{definition}[forthel,id=ClassOfPermsDef]
  Let $n$ be a natural number.
  $\emph{\Sym{n}}$ is the class of all permutations of $\natsegment{\one}{n}$.
\end{definition}
