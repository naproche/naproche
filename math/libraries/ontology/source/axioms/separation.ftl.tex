\begin{fakeforthel}
  \begin{axiom}[title=Separation Schema]
    Let $\tau$ be a term, $A$ be a class and $\varphi$ be a formula.
    If $A$ is a set then $\sepclass{\tau}{A}{\varphi}$ is a set.
  \end{axiom}
\end{fakeforthel}

\begin{fakeforthel}
  \begin{corollary}
    Let $A$ be a set and $B$ be a class.
    Assume that every element of $B$ is an element of $A$.
    Then $B$ is a set.
  \end{corollary}
\end{fakeforthel}
