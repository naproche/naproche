\begin{smodule}[title=Equality]{equality.ftl}

\symdef{eq}{\mathrel{\comp=}}
\symdef{neq}{\mathrel{\comp{\not=}}}

\STEXinvisible{\inlineforthel{Let $x \eq y$ stand for $x = y$.}}

\begin{fakeforthel}
  \begin{signature}[for={eq,neq}]
    Let $x,y$ be arbitrary entities.
    \emph{$x$ is equal to $y$} is a relation.

    Let $\emph{x \eq y}$ stand for $x$ is equal to $y$.
    Let $\emph{x \neq y}$ stand for $x$ is not equal to $y$.
  \end{signature}
\end{fakeforthel}

\begin{forthel}
  \begin{convention}
    Let $x$ and $y$ \emph{agree} stand for $x$ is equal to $y$.
    Let $x$ \emph{agrees with $y$} stand for $x$ is equal to $y$.
    Let $x$ and $y$ are \emph{distinct} stand for $x \neq y$.
  \end{convention}
\end{forthel}

\end{smodule}
