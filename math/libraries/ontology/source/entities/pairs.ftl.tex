\begin{forthel}
  \begin{convention}
    Let the \emph{ordered pair of $a$ and $b$} stand for $(a,b)$.
    Let an \emph{ordered pair} stand for an entity $x$ such that $x \eq (a,b)$ for some objects $a, b$.
  \end{convention}
\end{forthel}
