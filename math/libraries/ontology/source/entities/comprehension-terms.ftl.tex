\begin{smodule}[title=Comprehension Terms]{comprehension-terms.ftl}

\importmodule[libraries/ontology]{entities?classes.ftl}
\importmodule[libraries/ontology]{entities?objects.ftl}

\symdef{enumclass}[args=a]{\highlight\{\argsep{#1}{\highlight,}\highlight\}}
\symdef{compclass}[args=2]{\highlight\{#1\highlight\mid#2\highlight\}}
\symdef{sepclass}[args=3]{\highlight\{#1\highlight\in#2\highlight\mid#3\highlight\}}

\begin{fakeforthel}
  \begin{signature}
    Let $a_1,\dots,a_n$ be objects.
    $\emph{\enumclass{a1,\dots,a_n}}$ is a class.
  \end{signature}
\end{fakeforthel}

\begin{fakeforthel}
  \begin{signature}
    Let $\tau$ be a term and $\varphi$ be a formula.
    $\emph{\compclass{\tau}{\varphi}}$ is a class.
  \end{signature}
\end{fakeforthel}

\begin{fakeforthel}
  \begin{signature}
    Let $\tau$ be a term, $A$ be a class and $\varphi$ be a formula.
    $\emph{\sepclass{\tau}{A}{\varphi}}$ is a class.
  \end{signature}
\end{fakeforthel}

\end{smodule}
