\begin{smodule}[title=The Domain of a Map]{domain.ftl}

\importmodule[libraries/ontology]{entities?maps.ftl}
\importmodule[libraries/ontology]{entities?classes.ftl}

\symdef{dom}{\comp{\textrm{dom}}}

\begin{fakeforthel}
  \begin{signature}[for=dom]
    Let $f$ be a map.
    $\emph{\dom(f)}$ is a class.
  \end{signature}
\end{fakeforthel}

\begin{forthel}
  \begin{convention}
    Let the \emph{domain of $f$} stand for $\dom(f)$.
    Let $f$ is \emph{defined on $X$} stand for $\dom(f) = X$.
  \end{convention}
\end{forthel}

\end{smodule}
