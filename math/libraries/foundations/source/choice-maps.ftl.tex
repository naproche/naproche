\inlineforthel{[synonym map/-s]}

\begin{definition}[forthel,id=FOUNDATIONS_10_1352015487625913]
  Let $X$ be a system of nonempty sets.
  A \emph{choice map for $X$} is a map $g$ such that $\FUNdom(g) \eq X$ and $g(x) \SETin x$ for any $x \SETin X$.
\end{definition}
