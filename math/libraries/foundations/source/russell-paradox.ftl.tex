\begin{theorem}[forthel,title=Russell's Paradox,id=russell_paradox]
  Not every class is a set.
\end{theorem}
\begin{proof}[forthel]
  Assume the contrary.
  Define $R = \{x \mid x\text{ is a set such that }x \SETnotin x\}$.
  Then $R$ is a set.
  Hence $R \SETin R$ iff $R \SETnotin R$.
  Contradiction.
\end{proof}
