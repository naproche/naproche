\begin{definition}[forthel,id=FOUNDATIONS_14_2377279311183872]
  Let $X$ be a system of sets.
  $X$ is \emph{closed under countable unions} iff $\SETunionover U \SETin X$ for every nonempty countable subclass $U$ of $ X$.
\end{definition}
