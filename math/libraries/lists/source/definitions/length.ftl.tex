\documentclass{naproche-library}

\begin{document}
\begin{smodule}[title=Length]{length.ftl}

\importmodule[libraries/lists]{signatures?lists.ftl}
\importmodule[libraries/natural-numbers]{definitions?addition.ftl}
\importmodule[libraries/natural-numbers]{definitions?number-1.ftl}

\symdef{length}{\comp{\textsf{length}}}

\begin{signature}[forthel,id=LengthSig,for=length]
  Let $L$ be a list.
  $\emph{\length(L)}$ is a natural number.
\end{signature}

\begin{axiom}[forthel,id=LengthOfNilAx]
  $\length(\list{}) \eq \zero$.
\end{axiom}

\begin{axiom}[forthel,id=LengthOfConsAx]
  Let $a$ be an object and $L$ be a list.
  Then $\length(a \cons L) \eq \length(L) \plus \one$.
\end{axiom}

\end{smodule}
