\begin{smodule}[title=Snoc]{snoc.ftl}

\importmodule[libraries/lists]{definitions?concatenation.ftl}

\symdef{snoc}{\mathbin{\comp{::}}}

\begin{definition}[forthel,id=SnocDef,for=snoc]
  Let $L$ be a list and $a$ be an object.
  $\emph{L \snoc a} = L \concat \list{a}$.
\end{definition}
