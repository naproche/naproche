\begin{proposition}[forthel,id=LengthOfConcatProp]
  $\length(L \concat L') \eq \length(L) \plus \length(L')$ for all lists $L, L'$.
\end{proposition}
\begin{proof}[forthel,method=induction on $L$]
  Let $L$ be a list.

  \begin{case}{$L \eq \list{}$.} \end{case}

  \begin{case}{$L \eq a \cons L''$ for some object $a$ and some list $L''$.}
    Take an object $a$ and a list $L''$ such that $L \eq a \cons L''$.
    Then $L'' \prec L$.
    Hence $\length(L'' \concat L') \eq \length(L'') \plus \length(L')$ for every list $L'$.
    [prover vampire]
    Thus
    \begin{align*}
      \length(L \concat L')
        &\eq \length((a \cons L'') \concat L')\\
        &\eq \length(a \cons (L'' \concat L'))\\
        &\eq \length(L'' \concat L') \plus \one\\
        &\eq (\length(L'') \plus \length(L')) \plus \one\\
        &\eq (\length(L'') \plus \one) \plus \length(L')\\
        &\eq \length(a \cons L'') \plus \length(L')\\
        &\eq \length(L) \plus \length(L')
    \end{align*}
    for every list $L'$.
  \end{case}
\end{proof}
