\begin{smodule}[title=Recursive Maps]{recursive-maps.ftl}
\importmodule[libraries/set-theory]{ordering.ftl}
\symdef{ORDfunspace}[args=1]{#1^{\comp{<\infty}}}

\begin{definition}[forthel,id=SET_THEORY_04_1955917673267200]
  Let $A$ be a class.
  \[ \emph{\ORDfunspace{A}} = \{f \mid \text{$f$ is a map from $\alpha$ to $A$ for some ordinal $\alpha$} \}. \]
\end{definition}

\begin{proposition}[forthel,id=SET_THEORY_04_7841726894964736]
  Let $A$ be a class and $f$ be a map to $A$ such that $\dom(f)$ is a transitive subclass of $\Ord$ and $\alpha \in \dom(f)$.
  Then $f \FUNrest \alpha \in \ORDfunspace{A}$.
\end{proposition}

\inlineforthel{[prover vampire]}

\begin{definition}[forthel,id=SET_THEORY_04_5597213870784512]
  Let $H$ be a map and $G \FUNfromto{\ORDfunspace{A}}{A}$ for some class $A$ such that $H$ is a map to $A$.
  $H$ is \emph{recursive regarding $G$} iff $\dom(H)$ is a transitive subclass of $\Ord$ and for all $\alpha \in \dom(H)$ we have \[ H(\alpha) \eq G(H \FUNrest \alpha). \]
\end{definition}
