\begin{proposition}[forthel,id=hilbert_paradox_corollary]
  $\V$ is a proper class.
\end{proposition}
\begin{proof}[forthel]
  Assume the contrary.
  Then $\V$ is closed under powersets and closed under unions.
  Contradiction (by \nameref{hilbert_paradox}).
\end{proof}
