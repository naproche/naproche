\begin{theorem}[forthel,title=Burali-Forti's Paradox,id=burali_forti_paradox]
  $\Ord$ is a proper class.
\end{theorem}
\begin{proof}[forthel]
  Assume the contrary.
  $\Ord$ is transitive and every element of $\Ord$ is transitive.
  Hence $\Ord$ is an ordinal.
  Thus $\Ord \in \Ord$.
  Contradiction.
\end{proof}
