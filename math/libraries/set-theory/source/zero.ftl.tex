\begin{convention}[forthel]
  Let $\emph{0}$ stand for $\emptyset$.

  Let $\alpha$ is \emph{nonzero} stand for $\alpha \neq 0$.
\end{convention}

\begin{proposition}[forthel,id=SET_THEORY_02_8483196888940544]
  $0$ is an ordinal.
\end{proposition}
