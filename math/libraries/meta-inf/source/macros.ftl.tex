\begin{smodule}[title=Macros]{macros.ftl}
\symdef{eq}{\mathrel{\highlight=}}
\symdef{neq}{\mathrel{\highlight{\not=}}}
\symdef{elem}{\mathrel{\highlight\in}}
\symdef{nelem}{\mathrel{\highlight{\not\in}}}
\symdef{dom}{\highlight{\textrm{dom}}}
\symdef{iless}{\mathrel{\highlight\prec}}

\begin{forthel}
  Let $x \eq y$ stand for $x = y$.

  Let $x \elem y$ stand for $x \in y$.

  Let $x \notelem y$ stand for $x \notin y$.

  Let $x \iless y$ stand for $x \prec y$.
\end{forthel}
\end{smodule}
