\begin{theorem}[forthel,title=Cantor's Theorem,id=CantorProp]
  Let $x$ be a set.
  There exists no surjective map from $x$ onto $\pow(x)$.
\end{theorem}
\begin{proof}[forthel]
  Assume the contrary.
  Take a surjective map $f$ from $x$ onto $\pow(x)$.
  Define $C = \{ u \in x \mid u \nelem f(u) \}$.
  Then $C \elem \pow(x)$.
  Hence we can take a $u \elem x$ such that $f(u) \eq C$.
  Then $u \elem C$ iff $u \elem f(u)$ iff $u \nelem C$.
  Contradiction.
\end{proof}
