\begin{proposition}[forthel,id=IntersectsOverNonEmptySetsAreSetsProp]
  Let $X$ be a nonempty system of sets.
  Then $\intersectover X$ is a set.
\end{proposition}
\begin{proof}[forthel]
  Take an element $x$ of $X$.
  Then $\intersectover X \subclass x$.
  Hence $\intersectover X$ is a set.
\end{proof}
