\begin{proposition}[forthel,id=RangesOfFunsAreSetsProp]
  Let $f$ be a function.
  Then $\range(f)$ is a set.
\end{proposition}
\begin{proof}[forthel]
  $\range(f) \eq \im{f}{\dom(f)}$ and $\im{f}{\dom(f)}$ is a set.
  Hence $\range(f)$ is a set.
\end{proof}
