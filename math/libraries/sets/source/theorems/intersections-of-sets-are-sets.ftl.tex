\importmodule[libraries/sets]{axioms?separation.ftl}

\begin{proposition}[forthel,id=IntersectsOfSetsAreSetsProp]
  Let $x, y$ be sets.
  Then $x \intersect y$ is a set.
\end{proposition}
\begin{proof}[forthel]
  We have $x \intersect y \subclass x$.
  Hence $x \intersect y$ is a set .
\end{proof}
