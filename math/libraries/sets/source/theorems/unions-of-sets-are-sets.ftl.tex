\begin{proposition}[forthel,id=UnionsOfSetsAreSetsProp]
  Let $x, y$ be sets.
  Then $x \union y$ is a set.
\end{proposition}
\begin{proof}[forthel]
  Take $X = \pair{x}{y}$.
  Then $X$ is a set.
  Hence $\unionover X$ is a set.
  Indeed $X$ is a system of sets.
  We have $x \union y \eq \unionover X$.
  Thus $x \union y$ is a set.
\end{proof}
