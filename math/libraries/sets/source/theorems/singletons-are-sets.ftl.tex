\begin{smodule}{singletons-are-sets.ftl}

\importmodule[libraries/sets]{axioms?pairing.ftl}

\begin{proposition}[forthel,id=SingletonsAreSetsProp]
  Let $a$ be an object.
  Then $\singleton{a}$ is a set.
\end{proposition}

\begin{convention}[forthel]
  Let the singleton set of $a$ stand for the singleton class of $a$.
  Let a singleton set stand for a singleton class.
\end{convention}
