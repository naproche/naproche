\begin{definition}[forthel,id=MinimalElementDef]
  Let $X$ be a system of sets.
  An \emph{epsilon minimal element of $X$} is an element $x$ of $X$ such that
  for no $y \elem X$ we have $y \elem x$.
\end{definition}
