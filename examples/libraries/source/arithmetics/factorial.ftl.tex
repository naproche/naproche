  \section*{Factorial}

  \begin{forthel}
    \begin{signature}\printlabel{ARITHMETIC_12_0210357812531785}
      Let $n$ be a natural number.
      $n!$ is a natural number.
    \end{signature}
  \end{forthel}

  \begin{forthel}
    \begin{axiom}\printlabel{ARITHMETIC_12_6945210541236989}
      $0! = 1$.
    \end{axiom}
  \end{forthel}

  \begin{forthel}
    \begin{axiom}\printlabel{ARITHMETIC_12_2312015846987452}
      Let $n$ be a natural number.
      Then $(n + 1)! = n! \cdot (n + 1)$.
    \end{axiom}
  \end{forthel}

  \begin{forthel}
    \begin{proposition}\printlabel{ARITHMETIC_12_6123519005949952}
      Let $n$ be a natural number.
      Then $n! \neq 0$.
    \end{proposition}
    \begin{proof}
      Define $\Phi = \{n' \in \Nat \mid n'! \neq 0 \}$.

      (1) $\Phi$ contains $0$.
      Indeed $0! = 1 \neq 0$.

      (2) For all $n' \in \Phi$ we have $n' + 1 \in \Phi$. \\
      Proof.
        Let $n' \in \Phi$.
        We have $(n' + 1)! = (n' + 1) \cdot n'!$.
        $n' + 1$ and $n'!$ are nonzero.
        Hence $(n' + 1)! \neq 0$.
      Qed.

      Thus $\Phi$ contains every natural number (by \printref{ARITHMETIC_01_4764664342773760}).
      Therefore $n! \neq 0$.
    \end{proof}
  \end{forthel}
