  \section*{Closure Under Powersets}

  \begin{forthel}
    \begin{definition}[id=FOUNDATIONS_14_4652013578120594,printid]
      Let $X$ be a system of sets.
      $X$ is closed under powersets iff $\pow(U) \in X$ for every $U \in X$.
    \end{definition}
