  \section{Choice}

  \begin{forthel}
    \begin{axiom}[Axiom of Choice]\printlabel{FOUNDATIONS_10_1897613305577472}
      Let $X$ be a system of nonempty sets.
      Then there exists a map $f$ such that $\dom(f) = X$ and $f(x) \in x$ for any $x \in X$.
    \end{axiom}
  \end{forthel}
