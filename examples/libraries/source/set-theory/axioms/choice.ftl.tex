  \section{Choice}

  \begin{forthel}
    \begin{axiom}[title=Axiom of Choice,id=FOUNDATIONS_10_1897613305577472,printid]
      Let $X$ be a system of nonempty sets.
      Then there exists a map $f$ such that $\dom(f) = X$ and $f(x) \in x$ for any $x \in X$.
    \end{axiom}
  \end{forthel}
