  \section{Replacement}

  \begin{axiom}[forthel,title=Replacement Axiom,id=FOUNDATIONS_10_8142956584239104,printid]
    Let $f$ be a map and $x$ be a set.
    Then $f[x]$ is a set.
  \end{axiom}
