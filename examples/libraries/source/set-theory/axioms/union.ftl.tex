  \section{Union}

  \begin{axiom}[forthel,title=Union Axiom,id=FOUNDATIONS_10_5536459412996096,printid]
    Let $X$ be a system of sets.
    If $X$ is a set then $\bigcup X$ is a set.
  \end{axiom}
