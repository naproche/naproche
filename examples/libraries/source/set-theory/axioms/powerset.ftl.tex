  \section{Powerset}

  \begin{axiom}[forthel,title=Powerset Axiom,id=FOUNDATIONS_10_5862230203564032]
    Let $x$ be a set.
    Then $\pow(x)$ is a set.

    Let the powerset of $x$ stand for $\pow(x)$.
  \end{axiom}
