  \section*{Regular Cardinals}

  \begin{forthel}
    \begin{definition}[id=SET_THEORY_06_6532641205487950,printid]
      Let $\kappa$ be a cardinal.
      $\kappa$ is regular iff $|x| = \kappa$ for every cofinal subset $x$ of $\kappa$.
    \end{definition}
  \end{forthel}
