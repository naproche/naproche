  \section*{Weak Limit Cardinals}

  \begin{forthel}
    \begin{definition}[id=SET_THEORY_06_8945210236547450,printid]
      A weak limit cardinal is a cardinal number $\lambda$ such that $\lambda$ is not a successor cardinal and $\lambda \neq 0$.
    \end{definition}
