  \section*{Concatenation}

  \begin{forthel}
    \begin{signature}\printlabel{LISTS_CONCAT_4578620297183232}
      Let $L, L'$ be lists.
      $L \concat L'$ is a list.
    \end{signature}
  \end{forthel}

  \begin{forthel}
    \begin{axiom}\printlabel{LISTS_CONCAT_3703161885818880}
      Let $L$ be a list.
      Then $\nil \concat L = L$.
    \end{axiom}
  \end{forthel}

  \begin{forthel}
    \begin{axiom}\printlabel{LISTS_CONCAT_8050301789536256}
      Let $a$ be an object and $L, L'$ be lists.
      Then $(a \cons L) \concat L' = a \cons (L \concat L')$.
    \end{axiom}
  \end{forthel}

  \begin{forthel}
    \begin{proposition}\printlabel{LISTS_CONCAT_4512036658964875}
      $L \concat \nil = L$ for every list $L$.
    \end{proposition}
    \begin{proof}[by induction on $L$]
      Let $L$ be a list.

      Case $L = \nil$. Trivial.
      
      Case $L = a \cons L'$ for some object $a$ and some list $L'$.
        Consider an object $a$ and a list $L'$ such that $L = a \cons L'$.
        Then $L' \prec L$.
        Hence $L' \concat \nil = L'$.
        Thus $L \concat \nil
          = (a \cons L') \concat \nil
          = a \cons (L' \concat \nil)
          = a \cons L'
          = L$.
      End.
    \end{proof}
  \end{forthel}

  \begin{forthel}
    \begin{proposition}\printlabel{LISTS_CONCAT_1021563255448756}
      $L \concat (L' \concat L'') = (L \concat L') \concat L''$ for any lists $L, L', L''$.
    \end{proposition}
    \begin{proof}[by induction on $L$]
      Let $L$ be a list.

      Case $L = \nil$. Trivial.

      Case $L = a \cons L'''$ for some object $a$ and some list $L'''$.
        Consider an object $a$ and a list $L'''$ such that $L = a \cons L'''$.
        Then $L''' \prec L$.
        Let $L',L''$ be lists.
        Then $L''' \concat (L' \concat L'') = (L''' \concat L') \concat L''$.
        [prover vampire]
        Thus $L \concat (L' \concat L'') 
          = (a \cons L''') \concat (L' \concat L'')
          = a \cons (L''' \concat (L' \concat L''))
          = a \cons ((L''' \concat L') \concat L'')
          = (a \cons (L''' \concat L')) \concat L''
          = ((a \cons L''') \concat L') \concat L''
          = (L \concat L') \concat L''$.
      End.
    \end{proof}
  \end{forthel}
