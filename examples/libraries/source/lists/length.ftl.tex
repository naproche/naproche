\begin{smodule}{length.ftl}
  \importmodule[libraries]{lists?lists.ftl}
  \importmodule[libraries]{arithmetics?natural-numbers.ftl}

  \symdef{length}{\textrm{length}}

  \section*{Length}

  \begin{signature}[forthel,id=LISTS_LENGTH_4578620297183232,printid]
    Let $L$ be a list.
    $\length(L)$ is a natural number.
  \end{signature}

  \begin{axiom}[forthel,id=LISTS_LENGTH_3703161885818880,printid]
    $\length(\nil) = 0$.
  \end{axiom}

  \begin{axiom}[forthel,id=LISTS_LENGTH_8050301789536256,printid]
    Let $a$ be an object and $L$ be a list.
    Then $\length(a \cons L) = \length(L) + 1$.
  \end{axiom}
