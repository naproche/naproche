% E.g. `\set{1, 2, 3, 4}` prints the numbers 1, 2, 3 and 4 as a comma
% separated list which is enclosed within curly braces.
\def\set#1{%
  \ensuremath{%
    \{%
    \foreach[count=\i] \x in {#1}{%
      \ifnum\i>1,\,\fi%
      \x%
    }%
    \}
  }%
}
