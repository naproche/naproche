  \begin{fproposition*}[label=6125259604361216]
    Let $a, a', b, b'$ be objects.
    Assume $\upair{a}{b} = \upair{a'}{b'}$.
    Then ($a = a'$ and $b = b'$) or ($a = b'$ and $b = a'$).
  \end{fproposition*}
  \begin{fproof}
    We have $a = a'$ or $a = b'$.
    If $a = a'$ then $b = b'$.
    If $a = b'$ then $b = a'$.
    Hence ($a = a'$ and $b = b'$) or ($a = b'$ and $b = a'$).
  \end{fproof}
