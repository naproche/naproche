  \begin{forthel}
    \begin{theorem*}[Cantor's Theorem]\label{cantor}
      Let $x$ be a set.
      There exists no surjective map from $x$ onto $\pow(x)$.
    \end{theorem*}
    \begin{proof}
      Assume the contrary.
      Take a surjective map $f$ from $x$ onto $\pow(x)$.
      Define $C = \{ u \in x \mid u \notin f(u) \}$.
      Then $C \in \pow(x)$.
      Hence we can take a $u \in x$ such that $f(u) = C$.
      Then $u \in C$ iff $u \in f(u)$ iff $u \notin C$.
      Contradiction.
    \end{proof}
  \end{forthel}

  \begin{forthel}
    \begin{corollary*}
      Let $x$ be a set.
      Then $|x| < |\pow(x)|$.
    \end{corollary*}
    \begin{proof}
      Case $x$ is empty. Obvious.

      Case $x$ is nonempty.
        Assume the contrary.
        Then $|x| \geq |\pow(x)|$.
        Hence there exists a surjective map from $x$ onto $\pow(x)$ (by \printref{SET_THEORY_06_192336220913664}).
        Indeed $\pow(x)$ is a nonempty set.
        Contradiction (by \nameref{cantor}).
      End.
    \end{proof}
  \end{forthel}

  \begin{forthel}
    \begin{corollary*}
      For every ordinal $\alpha$ there exists a cardinal greater than $\alpha$.
    \end{corollary*}
    \begin{proof}
      Let $\alpha$ be an ordinal.
      Take $\kappa = |\pow(\alpha)|$.
      Then $\kappa > |\alpha|$.

      Let us show that $\kappa > \alpha$.
        Assume the contrary.
        Then $|\pow(\alpha)|
          = \kappa
          \leq \alpha$.
        Hence $\kappa
          = |\pow(\alpha)|
          = ||\pow(\alpha)||
          \leq |\alpha|$.
        Contradiction.
      End.
    \end{proof}
  \end{forthel}
